Python Programming

  • 1. Introduction to Python

    What is Python?

    Python is a high-level, interpreted programming language known for its simplicity and readability.

    Features:

    • Easy to learn and read
    • Large standard library
    • Dynamic typing
    • Object-oriented programming
  • 2. Basic Syntax

    Variables and Data Types

    # Variables
name = "John"    # String
age = 25        # Integer
height = 1.75   # Float
is_student = True  # Boolean

# Print variables
print(f"Name: {name}")
print(f"Age: {age}")
  • 3. Operators

    Common Operators

    # Arithmetic Operators
x = 10
y = 3
print(x + y)    # Addition: 13
print(x - y)    # Subtraction: 7
print(x * y)    # Multiplication: 30
print(x / y)    # Division: 3.333...
print(x % y)    # Modulus: 1
print(x ** y)   # Exponentiation: 1000

# Comparison Operators
print(x > y)    # True
print(x < y)    # False
print(x == y)   # False
print(x != y)   # True
  • 4. Control Structures

    If Statements and Loops

    # If-elif-else
age = 18
if age < 13:
    print("Child")
elif age < 20:
    print("Teenager")
else:
    print("Adult")

# For Loop
for i in range(5):
    print(i)

# While Loop
count = 0
while count < 5:
    print(count)
    count += 1

Java Programming

  • 1. Introduction to Java

    What is Java?

    Java is a class-based, object-oriented programming language designed to be platform-independent.

    Features:

    • Platform Independence (Write Once, Run Anywhere)
    • Object-Oriented
    • Rich Standard Library
    • Automatic Memory Management
  • 2. Basic Syntax

    Variables and Data Types

    public class BasicSyntax {
    public static void main(String[] args) {
        // Primitive Data Types
        int age = 25;
        double height = 1.75;
        boolean isStudent = true;
        char grade = 'A';

        // Reference Type
        String name = "John";
        
        System.out.println("Name: " + name);
        System.out.println("Age: " + age);
    }
}

    Common Operators

    public class Operators {
    public static void main(String[] args) {
        int x = 10;
        int y = 3;
        
        // Arithmetic Operators
        System.out.println(x + y);  // Addition: 13
        System.out.println(x - y);  // Subtraction: 7
        System.out.println(x * y);  // Multiplication: 30
        System.out.println(x / y);  // Division: 3
        System.out.println(x % y);  // Modulus: 1
        
        // Comparison Operators
        System.out.println(x > y);  // true
        System.out.println(x < y);  // false
        System.out.println(x == y); // false
        System.out.println(x != y); // true
    }
}
  • 4. Control Structures

    If Statements and Loops

    public class ControlStructures {
    public static void main(String[] args) {
        int age = 18;
        
        // If-else statement
        if (age < 13) {
            System.out.println("Child");
        } else if (age < 20) {
            System.out.println("Teenager");
        } else {
            System.out.println("Adult");
        }
        
        // For loop
        for (int i = 0; i < 5; i++) {
            System.out.println(i);
        }
        
        // While loop
        int count = 0;
        while (count < 5) {
            System.out.println(count);
            count++;
        }
    }
}

C++ Programming

  • 1. Introduction to C++

    What is C++?

    C++ is a powerful general-purpose programming language that extends C with object-oriented features.

    Features:

    • Object-Oriented Programming
    • Low-level Memory Manipulation
    • High Performance
    • Rich Standard Template Library (STL)
  • 2. Basic Syntax

    Variables and Data Types

    #include <iostream>
using namespace std;

int main() {
    // Basic Data Types
    int age = 25;
    double height = 1.75;
    bool isStudent = true;
    char grade = 'A';
    string name = "John";
    
    cout << "Name: " << name << endl;
    cout << "Age: " << age << endl;
    
    return 0;
}
  • 3. Operators

    Common Operators

    #include <iostream>
using namespace std;

int main() {
    int x = 10;
    int y = 3;
    
    // Arithmetic Operators
    cout << x + y << endl;  // Addition: 13
    cout << x - y << endl;  // Subtraction: 7
    cout << x * y << endl;  // Multiplication: 30
    cout << x / y << endl;  // Division: 3
    cout << x % y << endl;  // Modulus: 1
    
    // Comparison Operators
    cout << (x > y) << endl;  // 1 (true)
    cout << (x < y) << endl;  // 0 (false)
    cout << (x == y) << endl; // 0 (false)
    cout << (x != y) << endl; // 1 (true)
    
    return 0;
}
